    [docs]    def scopped_session(self, **kwargs):
        """
        Create scoped database session.
        """
        _kwargs = dict(rollback=self.rollback, fail=self.fail)
        _kwargs.update(**kwargs)
        try:
            with self._db.scoped_session(**_kwargs) as session:
                yield session
        finally:
            pass
